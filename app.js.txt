    function isWordLearned(level, lesson, idx) { return learnedWords.includes(`${level}_${lesson}_${idx}`); }
    function toggleWord(level, lesson, idx) {
        let key = `${level}_${lesson}_${idx}`;
        if (learnedWords.includes(key)) learnedWords = learnedWords.filter(k => k !== key);
        else learnedWords.push(key);
        localStorage.setItem('hsk_learned_full', JSON.stringify(learnedWords));
        if (currentTab === "study") renderCurrentLesson();
        updateStats();
    }

    function isLessonUnlocked(level, lesson) { return unlockedLessons[level]?.includes(lesson); }
    function unlockLesson(level, lesson) {
        if (!unlockedLessons[level]) unlockedLessons[level] = [];
        if (!unlockedLessons[level].includes(lesson)) {
            unlockedLessons[level].push(lesson);
            localStorage.setItem('hsk_unlocked_full', JSON.stringify(unlockedLessons));
        }
    }

    function getLessonLearnedCount(level, lesson) {
        return hskData[level][lesson].words.filter((_, i) => isWordLearned(level, lesson, i)).length;
    }

    function getTotalLearned() {
        let total = 0;
        for (let l of [1, 2, 3]) for (let i = 0; i < hskData[l].length; i++) total += getLessonLearnedCount(l, i);
        return total;
    }

    function getTotalWords() {
        let total = 0;
        for (let l of [1, 2, 3]) for (let les of hskData[l]) total += les.words.length;
        return total;
    }

    function updateStats() {
        let totalLearned = getTotalLearned(), totalWords = getTotalWords();
        document.getElementById('totalLearned').innerText = totalLearned;
        document.getElementById('currentLessonLearned').innerText = getLessonLearnedCount(currentLevel, currentLesson);
        document.getElementById('progressPercent').innerText = totalWords ? Math.round(totalLearned / totalWords * 100) + '%' : '0%';
    }

    function renderLessonNav() {
        let nav = document.getElementById('lessonNav');
        nav.innerHTML = '';
        for (let i = 0; i < hskData[currentLevel].length; i++) {
            let btn = document.createElement('button');
            btn.className = 'lesson-btn';
            if (i === currentLesson) btn.classList.add('active');
            if (getLessonLearnedCount(currentLevel, i) === hskData[currentLevel][i].words.length) btn.classList.add('learned');
            if (!isLessonUnlocked(currentLevel, i)) btn.classList.add('locked');
            btn.innerText = i + 1;
            btn.onclick = () => {
                if (isLessonUnlocked(currentLevel, i)) {
                    currentLesson = i;
                    renderLessonNav();
                    if (currentTab === "study") renderCurrentLesson();
                    else renderGames();
                }
            };
            nav.appendChild(btn);
        }
    }

    function renderCurrentLesson() {
        let lesson = hskData[currentLevel][currentLesson];
        let html = `<div class="words-grid">`;
        lesson.words.forEach((w, idx) => {
            let learned = isWordLearned(currentLevel, currentLesson, idx);
            html += `<div class="word-card">
                        <div class="chinese" onclick="speak('${w.chinese}','zh-CN')">${w.chinese} 🔊</div>
                        <div class="pinyin">${w.pinyin}</div>
                        <div class="meaning" onclick="speak('${w.meaning}','ru-RU')">${w.meaning} 🔊</div>
                        <button class="check-btn ${learned ? 'checked' : ''}" onclick="toggleWord(${currentLevel},${currentLesson},${idx})">${learned ? '✓' : '○'}</button>
                    </div>`;
        });
        let percent = getLessonLearnedCount(currentLevel, currentLesson) / lesson.words.length * 100;
        html += `</div><div class="lesson-progress"><div class="progress-bar"><div class="progress-fill" style="width:${percent}%"></div></div>
                 <button class="test-btn" onclick="startTest()">📝 Тест урока</button></div>`;
        document.getElementById('content').innerHTML = html;
        updateStats();
    }

    function renderGames() {
        let gameNames = ["🎯 Найди перевод", "🔊 Угадай по звуку", "✏️ Найди иероглиф", "📝 Составь пару"];
        let html = `<div class="game-container">
            <div class="game-selector">`;
        for (let i = 0; i < 4; i++) {
            html += `<button class="game-type-btn ${gameType === i ? 'active' : ''}" onclick="setGameType(${i})">${gameNames[i]}</button>`;
        }
        html += `</div><div id="gameContent"></div></div>`;
        document.getElementById('content').innerHTML = html;
        startNewGame();
    }

    function setGameType(type) {
        gameType = type;
        renderGames();
    }

    function startNewGame() {
        let lesson = hskData[currentLevel][currentLesson];
        let words = lesson.words.filter(w => w.chinese);
        if (words.length < 4) return;
        
        gameQuestions = [];
        for (let i = 0; i < 10; i++) {
            let correct = words[Math.floor(Math.random() * words.length)];
            let options = [correct];
            while (options.length < 4) {
                let opt = words[Math.floor(Math.random() * words.length)];
                if (!options.includes(opt)) options.push(opt);
            }
            options.sort(() => Math.random() - 0.5);
            
            let question, answer;
            if (gameType === 0) {
                question = { text: correct.chinese, type: "chinese" };
                answer = correct.meaning;
                options = options.map(o => o.meaning);
            } else if (gameType === 1) {
                question = { text: "🔊 Прослушай", audio: correct.chinese, type: "audio" };
                answer = correct.meaning;
                options = options.map(o => o.meaning);
            } else if (gameType === 2) {
                question = { text: correct.meaning, type: "meaning" };
                answer = correct.chinese;
                options = options.map(o => o.chinese);
            } else {
                question = { text: correct.chinese, type: "chinese" };
                answer = correct.meaning;
                options = options.map(o => o.meaning);
            }
            gameQuestions.push({ question, answer, options });
        }
        gameIndex = 0;
        gameScore = 0;
        gameAnswered = false;
        renderGameQuestion();
    }

    function renderGameQuestion() {
        if (gameIndex >= gameQuestions.length) {
            let percent = Math.round(gameScore / gameQuestions.length * 100);
            let modal = document.getElementById('gameContent');
            if (percent >= 70) showFireworks();
            else showPanda();
            modal.innerHTML = `<div class="game-result">
                <h2>${percent >= 70 ? '🎉 Молодец! 🎉' : '📚 Продолжай учиться!'}</h2>
                <p>Очки: ${gameScore} / ${gameQuestions.length}</p>
                <button class="next-game-btn" onclick="startNewGame()">🔄 Играть снова</button>
                <button class="next-game-btn" style="background:#667eea" onclick="renderGames()">🎮 Другие игры</button>
            </div>`;
            return;
        }
        
        let q = gameQuestions[gameIndex];
        let html = `<div class="game-score">Счёт: ${gameScore} / ${gameIndex} | Вопрос ${gameIndex+1}/${gameQuestions.length}</div>`;
        
        if (q.question.audio) {
            html += `<div class="game-question" onclick="speak('${q.question.audio}','zh-CN')">${q.question.text} 🔊</div>`;
            setTimeout(() => speak(q.question.audio, 'zh-CN'), 100);
        } else {
            html += `<div class="game-question">${q.question.text}</div>`;
        }
        
        html += `<div class="game-options">`;
        q.options.forEach(opt => {
            html += `<div class="game-option" onclick="checkGameAnswer('${opt.replace(/'/g, "\\'")}', this)">${opt}</div>`;
        });
        html += `</div>`;
        document.getElementById('gameContent').innerHTML = html;
        gameAnswered = false;
    }

    function checkGameAnswer(selected, element) {
        if (gameAnswered) return;
        gameAnswered = true;
        let correct = gameQuestions[gameIndex].answer;
        let allOpts = document.querySelectorAll('.game-option');
        allOpts.forEach(opt => {
            opt.style.pointerEvents = 'none';
            if (opt.innerText === correct) opt.classList.add('correct');
            if (opt === element && opt.innerText !== correct) opt.classList.add('wrong');
        });
        if (selected === correct) gameScore++;
        setTimeout(() => { gameIndex++; renderGameQuestion(); }, 800);
    }

    let currentTestQuestions = [], currentTestIndex = 0, testScore = 0;
    
    function startTest() {
        let lesson = hskData[currentLevel][currentLesson];
        let allWords = [...lesson.words];
        for (let i = allWords.length - 1; i > 0; i--) {
            let j = Math.floor(Math.random() * (i + 1));
            [allWords[i], allWords[j]] = [allWords[j], allWords[i]];
        }
        currentTestQuestions = allWords.slice(0, Math.min(10, allWords.length));
        currentTestIndex = 0;
        testScore = 0;
        renderTestQuestion();
        document.getElementById('testModal').classList.add('active');
    }
    
    function renderTestQuestion() {
        if (currentTestIndex >= currentTestQuestions.length) {
            let passed = testScore >= 9;
            if (passed && currentLesson + 1 < hskData[currentLevel].length) unlockLesson(currentLevel, currentLesson + 1);
            if (passed) showFireworks(); else showPanda();
            document.getElementById('testContent').innerHTML = `
                <div class="test-result">🎉 Результат: ${testScore}/${currentTestQuestions.length}<br>${testScore >= 9 ? '✅ Следующий урок открыт!' : '❌ Нужно 9/10'}</div>
                <button class="close-modal" onclick="closeTest()">Закрыть</button>
                ${testScore < 9 ? `<button class="close-modal" style="background:#f44336; margin-top:10px" onclick="closeTest(); startTest();">🔄 Ещё раз</button>` : ''}
            `;
            if (passed && currentLesson + 1 < hskData[currentLevel].length) { renderLessonNav(); if (currentTab === "study") renderCurrentLesson(); }
            return;
        }
        let q = currentTestQuestions[currentTestIndex];
        let allMeanings = hskData[currentLevel][currentLesson].words.map(w => w.meaning);
        let options = [q.meaning];
        while (options.length < 4) {
            let r = allMeanings[Math.floor(Math.random() * allMeanings.length)];
            if (!options.includes(r)) options.push(r);
        }
        options.sort(() => Math.random() - 0.5);
        let optsHtml = '';
        options.forEach(opt => {
            optsHtml += `<div class="test-option" onclick="checkTestAnswer('${opt.replace(/'/g, "\\'")}', this)">${opt}</div>`;
        });
        document.getElementById('testContent').innerHTML = `
            <div class="test-question" onclick="speak('${q.chinese}','zh-CN')">${q.chinese} 🔊</div>
            <div style="color:#e74c3c">${q.pinyin}</div>
            <div class="test-options">${optsHtml}</div>
            <div>${currentTestIndex+1}/${currentTestQuestions.length}</div>
            <div style="margin-top:10px">Нужно 9/10</div>
        `;
    }
    
    function checkTestAnswer(selected, element) {
        let correct = currentTestQuestions[currentTestIndex].meaning;
        let allOpts = document.querySelectorAll('.test-option');
        allOpts.forEach(opt => {
            opt.style.pointerEvents = 'none';
            if (opt.innerText === correct) opt.classList.add('correct');
            if (opt === element && opt.innerText !== correct) opt.classList.add('wrong');
        });
        if (selected === correct) testScore++;
        setTimeout(() => { currentTestIndex++; renderTestQuestion(); }, 800);
    }
