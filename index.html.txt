    <script>
        // При нажатии на уровень — загружаем приложение
        document.querySelectorAll('.level-card').forEach(card => {
            card.addEventListener('click', function() {
                const level = this.getAttribute('data-level');
                localStorage.setItem('selected_hsk_level', level);
                loadApp(level);
            });
        });

        function loadApp(level) {
            // Показываем загрузку
            document.body.innerHTML = `
                <div style="display: flex; justify-content: center; align-items: center; min-height: 100vh; color: white; font-size: 1.5rem;">
                    <div style="text-align: center;">
                        <div style="font-size: 3rem; margin-bottom: 20px;">📚</div>
                        <div>Загрузка HSK ${level}...</div>
                    </div>
                </div>
            `;
            
            // Загружаем app.js
            const script = document.createElement('script');
            script.src = 'app.js';
            script.onload = () => {
                if (window.initHSKApp) {
                    window.initHSKApp(parseInt(level));
                }
            };
            document.head.appendChild(script);
        }

        // Проверяем, нет ли уже сохранённого уровня
        const savedLevel = localStorage.getItem('selected_hsk_level');
        if (savedLevel && savedLevel !== 'undefined' && !window.location.search.includes('level')) {
            loadApp(savedLevel);
        }
    </script>
